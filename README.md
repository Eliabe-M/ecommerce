<!-- Ruy, se você abrir esse arquivo, entenda, seu trabalho nunca vai ser retornar nada para a página, mas sempre tem que retornar algo para mim 

server:

api: Contém os arquivos PHP que implementam os endpoints da API para diferentes recursos, como pedidos, produtos e usuários.
        orders: Manipula operações relacionadas a pedidos.
        products: Manipula operações relacionadas a produtos.
        users: Manipula operações relacionadas a usuários.
    db.php: Arquivo para configuração e gerenciamento de conexão com o banco de dados.

src:
    components: Componentes reutilizáveis do React.
        categorys.jsx: Componente para exibir categorias.
        navbar.jsx: Componente para a barra de navegação.
        searchBar.jsx: Componente para a barra de pesquisa.
    helpers: Funções de validação ou utilitários em PHP.
        loginValidator.php: Lógica para validar dados de login.
        productValidator.php: Lógica para validar dados de produto.
    models: Arquivos de modelo ou classes em PHP.
        product.php: Lógica relacionada ao modelo de produto.
        user.php: Lógica relacionada ao modelo de usuário
    services: Arquivos PHP que implementam serviços relacionados a autenticação e login.
        auth.php: Lógica relacionada à autenticação.
        login.php: Lógica relacionada ao login.
        logout.php: Lógica relacionada ao logout.
    views: Componentes de visualização do React.
        home.jsx: Página inicial do aplicativo.
        payment.jsx: Página de pagamento.
        product.jsx: Página de exibição de produto.
        products.jsx: Página de listagem de produtos.
        shoppingCart.jsx: Página de carrinho de compras.
        user.jsx: Página de perfil do usuário.

public:
    css: Arquivos de estilo.
    images: Imagens utilizadas no projeto.
    js: Arquivos JavaScript.
    index.php: Arquivo principal para renderizar a aplicação React no navegador.

-->
